special_project = {
	key = "TT_TERRAFORM_SOC"
	timelimit = -1
	cost = @tier4cost1
	tech_department = society_technology
	picture = GFX_evt_society_research
	icon = "gfx/interface/icons/situation_log/situation_log_society.dds"
	
	event_scope = planet_event
	
    abort_trigger = {
        OR = { 
            solar_system = { has_megastructure = dyson_sphere_3 } 
            solar_system = { has_megastructure = dyson_sphere_4 } 
            solar_system = { has_megastructure = dyson_sphere_5 }

            has_modifier = terraforming_candidate
            NOT = { has_modifier = tt_terraforming_project }

            NOR = {
                is_planet_class = pc_barren
                is_planet_class = pc_barren_cold
            }
        }
    }

	on_success = {
        remove_modifier = tt_terraforming_project
        add_modifier = { modifier = terraforming_candidate days = -1 }
    }
    on_cancel = {
        remove_modifier = tt_terraforming_project
    }
    on_fail = {
        remove_modifier = tt_terraforming_project
    }    
}
